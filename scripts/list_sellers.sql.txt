-- List vendor sellers for QA selection
--
-- If your WP table prefix is not "wp_", replace "wp_" everywhere.
-- This script resolves sellers from users/usermeta only (no seller_state lookup).
-- It returns only vendors who have BOTH products and orders.
-- Optimized for speed: pre-aggregated joins instead of correlated EXISTS.

SELECT
  u.ID AS user_id,
  u.user_login,
  u.user_email,
  u.display_name,

  MAX(CASE WHEN um.meta_key = 'billing_phone' THEN um.meta_value END) AS billing_phone,
  MAX(CASE WHEN um.meta_key = 'phone' THEN um.meta_value END) AS phone_meta,
  MAX(CASE WHEN um.meta_key = 'wcfm_phone' THEN um.meta_value END) AS wcfm_phone

FROM wp_users u
LEFT JOIN wp_usermeta um
  ON um.user_id = u.ID
  AND um.meta_key IN ('billing_phone', 'phone', 'wcfm_phone')
INNER JOIN (
  SELECT DISTINCT user_id
  FROM wp_usermeta
  WHERE meta_key = 'wp_capabilities'
    AND meta_value LIKE '%wcfm_vendor%'
) v ON v.user_id = u.ID
INNER JOIN (
  SELECT DISTINCT p.post_author AS seller_id
  FROM wp_posts p
  WHERE p.post_type = 'product'
    AND p.post_status NOT IN ('trash', 'auto-draft')
    AND p.post_author > 0
) sp ON sp.seller_id = u.ID
INNER JOIN (
  SELECT DISTINCT
    COALESCE(NULLIF(op.post_author, 0), NULLIF(parent_p.post_author, 0)) AS seller_id
  FROM wp_wc_order_product_lookup opl
  INNER JOIN wp_posts op ON op.ID = opl.product_id
  LEFT JOIN wp_posts parent_p ON parent_p.ID = op.post_parent
  WHERE COALESCE(NULLIF(op.post_author, 0), NULLIF(parent_p.post_author, 0)) IS NOT NULL
) so ON so.seller_id = u.ID

GROUP BY
  u.ID,
  u.user_login,
  u.user_email,
  u.display_name

ORDER BY u.ID DESC
LIMIT 200;

-- Quick count helper
SELECT COUNT(*) AS seller_rows
FROM (
  SELECT DISTINCT u.ID
  FROM wp_users u
  INNER JOIN (
    SELECT DISTINCT user_id
    FROM wp_usermeta
    WHERE meta_key = 'wp_capabilities'
      AND meta_value LIKE '%wcfm_vendor%'
  ) v ON v.user_id = u.ID
  INNER JOIN (
    SELECT DISTINCT p.post_author AS seller_id
    FROM wp_posts p
    WHERE p.post_type = 'product'
      AND p.post_status NOT IN ('trash', 'auto-draft')
      AND p.post_author > 0
  ) sp ON sp.seller_id = u.ID
  INNER JOIN (
    SELECT DISTINCT
      COALESCE(NULLIF(op.post_author, 0), NULLIF(parent_p.post_author, 0)) AS seller_id
    FROM wp_wc_order_product_lookup opl
    INNER JOIN wp_posts op ON op.ID = opl.product_id
    LEFT JOIN wp_posts parent_p ON parent_p.ID = op.post_parent
    WHERE COALESCE(NULLIF(op.post_author, 0), NULLIF(parent_p.post_author, 0)) IS NOT NULL
  ) so ON so.seller_id = u.ID
) x;
