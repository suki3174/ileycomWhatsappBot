-- Verify seller phone rebind for auth/products/orders flows
--
-- Usage:
-- 1) Set exactly one identity: @seller_id OR @seller_login OR @seller_email
-- 2) Set @expected_phone to the phone you rebound
-- 3) If your WP prefix is not "wp_", replace it everywhere

SET @seller_id := NULL;
SET @seller_login := 'cwsb_test_seller';
SET @seller_email := NULL;
SET @expected_phone := '21650111222';

SET @resolved_user_id := COALESCE(
  NULLIF(@seller_id, 0),
  (SELECT u.ID FROM wp_users u WHERE BINARY u.user_login = BINARY @seller_login LIMIT 1),
  (SELECT u.ID FROM wp_users u WHERE BINARY u.user_email = BINARY @seller_email LIMIT 1)
);

-- 1) Basic identity
SELECT
  u.ID AS user_id,
  u.user_login,
  u.user_email,
  u.display_name
FROM wp_users u
WHERE u.ID = @resolved_user_id;

-- 2) Phone fields + pass/fail
SELECT
  @resolved_user_id AS user_id,
  MAX(CASE WHEN um.meta_key = 'billing_phone' THEN um.meta_value END) AS billing_phone,
  MAX(CASE WHEN um.meta_key = 'phone' THEN um.meta_value END) AS phone_meta,
  MAX(CASE WHEN um.meta_key = 'wcfm_phone' THEN um.meta_value END) AS wcfm_phone,
  CASE
    WHEN
      BINARY MAX(CASE WHEN um.meta_key = 'billing_phone' THEN um.meta_value END) = BINARY @expected_phone
      AND BINARY MAX(CASE WHEN um.meta_key = 'phone' THEN um.meta_value END) = BINARY @expected_phone
      AND BINARY MAX(CASE WHEN um.meta_key = 'wcfm_phone' THEN um.meta_value END) = BINARY @expected_phone
    THEN 'PASS'
    ELSE 'FAIL'
  END AS phone_rebind_status
FROM wp_usermeta um
WHERE um.user_id = @resolved_user_id
  AND um.meta_key IN ('billing_phone', 'phone', 'wcfm_phone');

-- 3) Seller state check (auth flow linkage)
SELECT
  s.user_id,
  s.phone AS state_phone,
  s.flow_token,
  s.session_active_until,
  CASE WHEN BINARY s.phone = BINARY @expected_phone THEN 'PASS' ELSE 'FAIL' END AS state_phone_status,
  CASE WHEN s.flow_token IS NULL THEN 'PASS' ELSE 'WARN_NOT_NULL' END AS flow_token_reset_status
FROM wp_cwsb_seller_state s
WHERE s.user_id = @resolved_user_id;

-- 4) Products owned by seller (products flow)
SELECT
  COUNT(*) AS product_count
FROM wp_posts p
WHERE p.post_type = 'product'
  AND p.post_author = @resolved_user_id
  AND p.post_status NOT IN ('trash', 'auto-draft');

SELECT
  p.ID AS product_id,
  p.post_title,
  p.post_status
FROM wp_posts p
WHERE p.post_type = 'product'
  AND p.post_author = @resolved_user_id
ORDER BY p.ID DESC;

-- 5) Orders linked by expected phone (orders flow sanity)
SELECT
  COUNT(DISTINCT pm.post_id) AS orders_with_expected_phone
FROM wp_postmeta pm
JOIN wp_posts o ON o.ID = pm.post_id
WHERE o.post_type = 'shop_order'
  AND pm.meta_key = '_billing_phone'
  AND BINARY pm.meta_value = BINARY @expected_phone;

SELECT
  pm.post_id AS order_id,
  pm.meta_value AS billing_phone,
  o.post_status
FROM wp_postmeta pm
JOIN wp_posts o ON o.ID = pm.post_id
WHERE o.post_type = 'shop_order'
  AND pm.meta_key = '_billing_phone'
  AND BINARY pm.meta_value = BINARY @expected_phone
ORDER BY pm.post_id DESC;
